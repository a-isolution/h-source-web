"use client";

import React, { useState, useEffect } from "react";
import { ArrowRight, UploadCloud, X } from "lucide-react";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { Label } from "@/components/ui/label";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import Spinner from "@/components/spinner";
import AddBrandModal from "@/components/product/add-brand-modal";
import AddCategoryModal from "@/components/product/add-category-modal";
import {
  useCreateProduct,
  useGetProductById,
  useUpdateProduct,
} from "@/api/products";
import { useGetBrands } from "@/api/brands";
import { useGetCategories } from "@/api/category";
import { useGetCurrencies } from "@/api/currency";
import { useQueryClient } from "@tanstack/react-query";
import { toast } from "sonner";
import { useRouter, useSearchParams } from "next/navigation";

const CreateProduct = () => {
  const searchParams = useSearchParams();
  const productId = searchParams.get("id");
  const qc = useQueryClient();
  const router = useRouter();

  const { data } = useGetProductById({ productId });
  const { data: currencies } = useGetCurrencies();
  const { data: brands } = useGetBrands();
  const { data: categories } = useGetCategories();
  const { mutate, isPending } = useCreateProduct();
  const { mutate: editProduct, isPending: updateIsPending } =
    useUpdateProduct();

  const [name, setName] = useState("");
  const [price, setPrice] = useState("");
  const [description, setDescription] = useState("");
  const [stockQuantity, setStockQuantity] = useState("");
  const [images, setImages] = useState<File[]>([]);
  const [currencyId, setCurrencyId] = useState("");
  const [brandId, setBrandId] = useState("");
  const [categoryId, setCategoryId] = useState("");
  const [sizeId, setSizeId] = useState("");
  const [isActive, setIsActive] = useState<number>(data?.isActive ?? 1);
  const [imagePreviews, setImagePreviews] = useState<string[]>([]); // Added state for image previews

  useEffect(() => {
    if (data && productId) {
      setName(data.name || "");
      setDescription(data.description || "");
      setStockQuantity(data.stockQuantity?.toString() || "");
      setCurrencyId(data.currencyId?.toString() || "");
      setBrandId(data.brandId?.toString() || "");
      setCategoryId(data.categoryId?.toString() || "");
      setIsActive(typeof data.isActive === "number" ? data.isActive : 1);

      if (typeof data.price === "number") {
        setPrice((data.price / 100).toFixed(2));
      } else {
        setPrice("");
      }

      setImages([]);
      setImagePreviews([]); // Reset image previews when editing a product
    }
  }, [data, productId]);

  const handleImageUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files) {
      const selectedFiles = Array.from(e.target.files);
      setImages((prev) => [...prev, ...selectedFiles]);

      // Generate preview URLs for the selected images
      const imagePreviews = selectedFiles.map((file) =>
        URL.createObjectURL(file)
      );
      setImagePreviews((prev) => [...prev, ...imagePreviews]);
    }
  };

  const removeImage = (index: number) => {
    setImages((prev) => prev.filter((_, i) => i !== index));
    setImagePreviews((prev) => prev.filter((_, i) => i !== index)); // Remove preview as well
  };

  const handleSubmit = () => {
    const priceRegex = /^\d+(\.\d{1,2})?$/;
    if (!priceRegex.test(price)) {
      alert("Please enter a valid price with up to 2 decimal places.");
      return;
    }

    const priceFloat = parseFloat(price);
    const priceInt = Math.round(priceFloat * 100);

    const formData = new FormData();
    formData.append("name", name);
    formData.append("description", description);
    formData.append("price", priceInt.toString());
    formData.append("currencyId", currencyId);
    formData.append("brandId", brandId);
    formData.append("categoryId", categoryId);
    formData.append("isFeatured", "1");
    formData.append("isActive", isActive.toString());
    formData.append("stockQuantity", stockQuantity);

    images.forEach((file) => {
      formData.append("files", file);
    });

    if (productId) {
      editProduct(
        { body: formData as any, productId },
        {
          onSuccess: (data) => {
            toast.success(data?.message || "Product updated successfully!");
            qc.invalidateQueries({ queryKey: ["my-products"] });
          },
        }
      );
    } else {
      mutate(
        { body: formData as any },
        {
          onSuccess: (data) => {
            toast.success(data?.message || "Product created successfully!");
            qc.invalidateQueries({ queryKey: ["my-products"] });
            router.push("/products");
            // Clear form after submission
            setName("");
            setPrice("");
            setDescription("");
            setStockQuantity("");
            setImages([]);
            setCurrencyId("");
            setBrandId("");
            setCategoryId("");
            setIsActive(1);
            setImagePreviews([]);
          },
        }
      );
    }
  };

  const [openBrandModal, setOpenBrandModal] = useState(false);
  const handleAddBrand = () => setOpenBrandModal(true);

  const [openCategoryModal, setOpenCategoryModal] = useState(false);
  const handleAddCategory = () => setOpenCategoryModal(true);

  return (
    <div className="p-2 sm:p-6 space-y-6">
      <div>
        <h1 className="text-xl font-medium">Products</h1>
        <span className="text-sm text-[#A7A7A7]">
          Manage your drinks and candy inventory
        </span>
      </div>

      {productId && (
        <div className="flex justify-end">
          <Button
            onClick={handleSubmit}
            className="bg-lime-400 min-w-32 text-black px-6 py-2 rounded-sm font-medium flex items-center gap-2 hover:bg-lime-500 transition"
          >
            {updateIsPending ? <Spinner /> : "Save Changes"}
          </Button>
        </div>
      )}

      <div className="grid grid-cols-1 md:grid-cols-2 gap-4 sm:gap-6">
        <div className="space-y-8">
          <div className="space-y-2">
            <Label>Name</Label>
            <Input
              value={name}
              onChange={(e) => setName(e.target.value)}
              placeholder="Product name"
            />
          </div>

          <div className="space-y-2">
            <Label>Brand</Label>
            <div className="flex flex-row items-center gap-2">
              <Select
                value={brandId}
                onValueChange={(value) => setBrandId(value)}
              >
                <SelectTrigger className="w-full">
                  <SelectValue placeholder="Select a brand" />
                </SelectTrigger>
                <SelectContent>
                  {brands?.data?.map((brand: any) => (
                    <SelectItem key={brand.id} value={brand.id.toString()}>
                      {brand.name}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
              <Button
                onClick={handleAddBrand}
                variant="secondary"
                size="sm"
                className="flex items-center gap-1 w-fit"
              >
                Add new
              </Button>
            </div>
          </div>

          <div className="space-y-2">
            <Label>Category</Label>
            <div className="flex flex-row items-center gap-2">
              <Select
                value={categoryId}
                onValueChange={(value) => setCategoryId(value)}
              >
                <SelectTrigger className="w-full">
                  <SelectValue placeholder="Select a category" />
                </SelectTrigger>
                <SelectContent>
                  {categories?.data?.map((cat: any) => (
                    <SelectItem key={cat.id} value={cat.id.toString()}>
                      {cat.name}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
              <Button
                onClick={handleAddCategory}
                variant="secondary"
                size="sm"
                className="flex items-center gap-1 w-fit"
              >
                Add new
              </Button>
            </div>
          </div>

          <div className="space-y-2">
            <Label>Description</Label>
            <textarea
              rows={6}
              value={description}
              onChange={(e) => setDescription(e.target.value)}
              placeholder="Write a product description..."
              className="w-full p-3 rounded-md border border-gray-300"
            />
          </div>
        </div>

        <div className="space-y-8">
          <div className="space-y-2">
            <Label>Price</Label>
            <Input
              value={price}
              onChange={(e) => setPrice(e.target.value)}
              placeholder="Price in USD"
              type="number"
              step="0.01"
            />
          </div>

          <div className="space-y-2">
            <Label>Stock Quantity</Label>
            <Input
              value={stockQuantity}
              onChange={(e) => setStockQuantity(e.target.value)}
              placeholder="Enter stock quantity"
              type="number"
            />
          </div>

          <div className="space-y-2">
            <Label>Upload Images</Label>
            <div className="w-full border-2 border-dashed rounded-md bg-gray-50 cursor-pointer relative">
              <Input
                type="file"
                accept="image/*"
                multiple
                onChange={handleImageUpload}
                className="hidden"
                id="upload"
              />
              <label
                htmlFor="upload"
                className="flex flex-col items-center justify-center h-48"
              >
                <UploadCloud className="w-8 h-8 text-gray-400" />
                <span className="text-gray-500 text-sm mt-2">
                  {images.length > 0
                    ? `${images.length} image${
                        images.length > 1 ? "s" : ""
                      } selected`
                    : "Click to upload images"}
                </span>
              </label>
            </div>
          </div>

          {imagePreviews.length > 0 && (
            <div className="mt-2 grid grid-cols-2 gap-2 sm:grid-cols-3 lg:grid-cols-4">
              {imagePreviews.map((preview, index) => (
                <div key={index} className="relative">
                  <img
                    src={preview}
                    alt={`Image preview ${index + 1}`}
                    className="object-cover w-full h-32 rounded-md"
                  />
                  <button
                    onClick={() => removeImage(index)}
                    className="absolute top-1 right-1 text-red-500 hover:text-red-700"
                    title="Remove"
                  >
                    <X className="w-5 h-5" />
                  </button>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>

      <div className="flex justify-center mt-8">
        <Button
          onClick={handleSubmit}
          className="bg-lime-400 min-w-32 text-black px-6 py-2 rounded-sm font-medium flex items-center gap-2 hover:bg-lime-500 transition"
        >
          {isPending || updateIsPending ? <Spinner /> : "Save Product"}
        </Button>
      </div>
    </div>
  );
};

export default CreateProduct;
